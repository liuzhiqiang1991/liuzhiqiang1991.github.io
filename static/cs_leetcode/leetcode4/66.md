    class Solution {
        public int[] plusOne(int[] digits) {
            int rest = 1;
            for(int i = digits.length - 1; i >= 0; i--) {
                if(rest == 0) break;
                rest = rest + digits[i];
                digits[i] = rest % 10;
                rest = rest / 10;
            }
            if(rest != 0) {
                int[] res = new int[digits.length + 1];
                res[0] = rest;
                for(int i = 0; i <= digits.length - 1; i++) {
                    res[i + 1] = digits[i];
                }
                return res;
            }
            return digits;
        }
    }

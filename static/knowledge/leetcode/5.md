### 5.最长回文子串

```
    class Solution {
        public String longestPalindrome(String s) {
            String ret = "";
            int max = Integer.MIN_VALUE;
            char[] c = s.toCharArray();
            for(int i = 0; i <= c.length - 1; i++) {
                int left = i;
                int right = i;
                while(left >= 0 && right <= c.length - 1) {
                    if(c[left] == c[right]) {
                        if(right - left + 1 > max) {
                            max = right - left + 1;
                            ret = s.substring(left, right + 1);
                        }
                    } else {
                        break;
                    }
                    left--;
                    right++;
                }
            }

            for(int i = 0; i <= c.length - 1; i++) {
                int left = i;
                int right = i + 1;
                while(left >= 0 && right <= c.length - 1) {
                    if(c[left] == c[right]) {
                        if(right - left + 1 > max) {
                            max = right - left + 1;
                            ret = s.substring(left, right + 1);
                        }
                    } else {
                        break;
                    }
                    left--;
                    right++;
                }
            }
            return ret;
        }
    }
```

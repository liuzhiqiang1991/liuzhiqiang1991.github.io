#### 83.删除排序链表中的重复元素
```
    /**
     * Definition for singly-linked list.
     * public class ListNode {
     *     int val;
     *     ListNode next;
     *     ListNode() {}
     *     ListNode(int val) { this.val = val; }
     *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
     * }
     */
    class Solution {
        public ListNode deleteDuplicates(ListNode head) {
            if(head == null) return null;
            ListNode first = head;
            ListNode second = head.next;
            Set<Integer> set = new HashSet<>();
            set.add(head.val);
            while(second != null) {
                if(set.contains(second.val)) {
                    second = second.next;
                    first.next = second;
                } else {
                    set.add(second.val);
                    second = second.next;
                    first = first.next;
                }
            }
            return head;
        }
    }
```

#### 方法二
```
    class Solution {
        public ListNode deleteDuplicates(ListNode head) {
            ListNode pre = head;
            ListNode cur = pre;
            while(cur != null) {
                int val = cur.val;
                while(cur.next != null && cur.next.val == val) {
                    cur = cur.next;
                }
                pre.next = cur.next;
                pre = pre.next;
                cur = pre;
            }
            return head;
        }
    }
```

#### 994. 腐烂的橘子
```
    class Solution {
        public int orangesRotting(int[][] grid) {
            int m = grid.length;
            int n = grid[0].length;
            int res = 0;
            while(true) {
                int c1 = count(grid, m, n);
                rotting(grid, m, n);
                int c2 = count(grid, m, n);
                if(c1 == c2) break;
                res++;
            }
            for(int i = 0; i <= m - 1; i++) {
                for(int j = 0; j <= n - 1; j++) {
                    if(grid[i][j] == 1) return -1;
                }
            }
            return res;
        }

        public int count(int[][] grid, int m, int n) {
            int count = 0;
            for(int i = 0; i <= m - 1; i++) {
                for(int j = 0; j <= n - 1; j++) {
                    if(grid[i][j] == 2) count++;
                }
            }
            return count;
        }

        public void rotting(int[][] grid, int m, int n) {
            for(int i = 0; i <= m - 1; i++) {
                for(int j = 0; j <= n - 1; j++) {
                    if(grid[i][j] == 2) {
                        rot(grid, m, n, i - 1, j);
                        rot(grid, m, n, i + 1, j);
                        rot(grid, m, n, i, j - 1);
                        rot(grid, m, n, i, j + 1);
                    }
                }
            }
            for(int i = 0; i <= m - 1; i++) {
                for(int j = 0; j <= n - 1; j++) {
                    if(grid[i][j] == 3) grid[i][j] = 2;
                }
            }
        }

        public void rot(int[][] grid, int m, int n, int i, int j) {
            if(i < 0 || i > m - 1 || j < 0 || j > n - 1) return;
            if(grid[i][j] == 0) return;
            if(grid[i][j] == 1) grid[i][j] = 3;
        }
    }
```

#### 160. 相交链表
```
    public class Solution {
        public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
            ListNode n1 = headA;
            int c1 = 0;
            while(n1 != null) {
                n1 = n1.next;
                c1++;
            }
            ListNode n2 = headB;
            int c2 = 0;
            while(n2 != null) {
                n2 = n2.next;
                c2++;
            }
            n1 = headA;
            n2 = headB;
            if(c1 > c2) {
                int d = c1 - c2;
                for(int i = 1; i <= d; i++) n1 = n1.next;
            } else {
                int d = c2 - c1;
                for(int i = 1; i <= d; i++) n2 = n2.next;
            }
            while(n1 != null && n2 != null && n1 != n2) {
                n1 = n1.next;
                n2 = n2.next;
            }
            if(n1 != null && n2 != null) return n1;
            else return null;
        }
    }
```

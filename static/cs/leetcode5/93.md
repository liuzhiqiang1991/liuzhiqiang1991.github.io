#### 93.复原 IP 地址
```
    class Solution {
        public List<String> restoreIpAddresses(String s) {
            List<String> res = new ArrayList<>();
            if(s.length() >= 13 || s.length() <= 3) return res;

            for(int i = 1; i <= 3; i++) {
                for(int j = 1; j <= 3; j++) {
                    for(int k = 1; k <= 3; k++) {
                        if(i + j + k > s.length() - 1) continue;
                        String n1 = s.substring(0, i);
                        String n2 = s.substring(i, i + j);
                        String n3 = s.substring(i + j, i + j + k);
                        String n4 = s.substring(i + j + k);
                        if(check(n1, n2, n3, n4)) {
                            StringBuilder sb = new StringBuilder();
                            sb.append(n1).append(".").append(n2).append(".").append(n3).append(".").append(n4);
                            res.add(sb.toString());
                        }
                    }
                }
            }
            return res;
        }

        public boolean check(String n1, String n2, String n3, String n4) {
            if("".equals(n1) || "".equals(n2) || "".equals(n3) || "".equals(n4)) {
                return false;
            }
            if((n1.startsWith("0") && !n1.equals("0"))
                || (n2.startsWith("0") && !n2.equals("0"))
                || (n3.startsWith("0") && !n3.equals("0"))
                || (n4.startsWith("0") && !n4.equals("0"))) {
                return false;
            }
            if(Integer.valueOf(n1) > 255 || Integer.valueOf(n2) > 255 || Integer.valueOf(n3) > 255 || Integer.valueOf(n4) > 255) {
                return false;
            }
            return true;
        }

    }
```

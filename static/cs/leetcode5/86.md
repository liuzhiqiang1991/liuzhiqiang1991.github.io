#### 86.分隔链表

#### 不在原链表上改
```
    class Solution {
        public ListNode partition(ListNode head, int x) {
            ListNode first = new ListNode(0, null);
            ListNode second = new ListNode(0, null);
            ListNode node = head;
            ListNode f1 = first;
            ListNode s1 = second;
            while(node != null) {
                ListNode tmp = new ListNode(node.val, null);
                if(node.val < x) {
                    f1.next = tmp;
                    f1 = f1.next;
                } else {
                    s1.next = tmp;
                    s1 = s1.next;
                }
                node = node.next;
            }
            f1.next = second.next;
            return first.next;
        }
    }
```

#### 在原链表上改

```
    class Solution {
        public ListNode partition(ListNode head, int x) {
            ListNode first = new ListNode(0, null);
            ListNode second = new ListNode(0, null);
            ListNode f1 = first;
            ListNode s1 = second;
            ListNode node = head;
            while(node != null) {
                if(node.val < x) {
                    f1.next = node;
                    f1 = f1.next;
                } else {
                    s1.next = node;
                    s1 = s1.next;
                }
                node = node.next;
            }
            s1.next = null;
            f1.next = second.next;
            return first.next;
        }
    }
```
